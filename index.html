function onScanSuccess(barcode) { html5QrCode.pause(true); if (navigator.vibrate) navigator.vibrate(50); document.getElementById('loading').style.display = 'block'; document.getElementById('scanResultSection').style.display = 'none'; fetch(GAS_URL, { method: "POST", body: JSON.stringify({ action: "getProduct", barcode: barcode }) }) .then(res => res.json()) .then(product => { document.getElementById('loading').style.display = 'none'; if (product && product.name) { tempProduct = product; // 暫存產品資料 document.getElementById('scanResultSection').style.display = 'block'; const qtyInput = document.getElementById('pQty'); // 判斷是否為時價產品 (價格為 0) if (product.price === 0) { document.getElementById('pName').innerText = product.name + " (請輸入總價)"; document.getElementById('pUnit').innerText = "£ 總額"; qtyInput.placeholder = "輸入成交總金額"; qtyInput.value = ""; } else { document.getElementById('pName').innerText = product.name + " (£" + product.price + ")"; document.getElementById('pUnit').innerText = "數量"; qtyInput.value = "1"; } qtyInput.focus(); } else { alert("找不到產品！"); html5QrCode.resume(); } }); } function addToCart() { const inputVal = parseFloat(document.getElementById('pQty').value); if (isNaN(inputVal) || inputVal <= 0) return alert("請輸入有效數值"); let finalSubtotal = 0; let displayQty = inputVal; if (tempProduct.price === 0) { // 時價產品：輸入值直接就是總額 finalSubtotal = inputVal; displayQty = 1; // 顯示為 1 份 } else { // 普通產品：數量 * 單價 finalSubtotal = inputVal * tempProduct.price; } cart.push({ name: tempProduct.name, qty: displayQty, unit: (tempProduct.price === 0) ? "份" : "個", subtotal: finalSubtotal }); renderCart(); document.getElementById('scanResultSection').style.display = 'none'; tempProduct = null; setTimeout(() => { html5QrCode.resume(); }, 500); }
